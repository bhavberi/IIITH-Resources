#include <stdio.h>
#include <stdlib.h>
#include <ctype.h>
#include <string.h>
#include <math.h>

int main() 
{
	int N,x;
	x=scanf("%d",&N);  //gets the number of triplets to be printed
	if(x<0)
		{
			return 1;
		}
	if(N<=0)  //checks for user error
		{
			return 1;
		}
	int n=0, a, b, c;
	for(a=3;a<=200;a++)  //the following nested loops are used to increment each value sequentially
		{
			for(b=a+1;b<=200;b++)  //using b=a+1 makes sure a>b and then b>c in next loop
				{
					for(c=b+1;c<=200;c++)
						{
							if((a*a)+(b*b)==(c*c))  //checks the condition for triplets
								{
									printf("%d %d %d",a,b,c);
									n++;
									break;  //no need of c loop anymore 
								}
						}
					if(n==N)  //no need if we already got what we needed
						{
							break;
						}
				}
			if(n==N)  //again don't be greedy, you got what you wanted :D
				{
					break;
				}
		}
}
